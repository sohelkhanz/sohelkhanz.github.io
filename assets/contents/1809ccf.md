```python
pi = lambda x: print(x,end=" ")
ii = lambda: int(input())
ml = lambda: list(map(int,input().split()))
for _ in [0]*ii():
	n,k= ml()
	x = ((8*k +1)**.5 -1)//2
	y = k - x*(x+1)/2
	z = 2*(x-y) + 1
	for i in range(n):
		if i<x:
			pi(2)
		elif i==x:
			pi(int(-z))
		else:
			pi(-1000)
	print()
```
